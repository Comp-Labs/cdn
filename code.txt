#include <LiquidCrystal.h>
LiquidCrystal LCD(7,8,9,10,11,12);
int TRIG = 4;
int ECHO = 5;

void setup() {
  pinMode(TRIG,OUTPUT);
  pinMode(ECHO,INPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(TRIG,HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG,LOW);
  int TIME = pulseIn(ECHO,HIGH);
  int DISTANCE = (0.034*TIME/2);
  Serial.println(DISTANCE);
  LCD.setCursor(0,0);
  LCD.print(DISTANCE);
  delay(0.01);
  LCD.clear();
}
